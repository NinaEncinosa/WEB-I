>>>>>
>>>>>html{font-family: "MS Serif"; color: black;}
>>>>>
>>>>>h1,h2,h3{color: red; font-weight: bold; }
>>>>>
>>>>>h4,h5{ color: gray; }

>>>>>article p { font-size: 14px; }
>>>>>
>>>>>article p a { color: green; }
>>>>>
>>>>>article p a:hover { color: black;}
